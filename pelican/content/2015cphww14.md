    ctx.beginPath()
    ctx.lineWidth = 5
    ctx.moveTo(0,0+i*20 )
    ctx.lineTo(200,0+i*20 )
    ctx.strokeStyle = "#0000ff"
    ctx.stroke()

    ctx.lineWidth = 5

    ctx.strokeStyle = "#FF0000"
    ctx.stroke()


</script>
